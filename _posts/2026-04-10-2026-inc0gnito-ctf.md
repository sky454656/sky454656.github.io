### A. 스택 카나리 오염 (The Trigger)

v4 버퍼는 rbp-0x20에서 시작하고, 카나리는 rbp-0x8에 있습니다. 이 사이의 거리는 24바이트입니다.

- 하지만 iovec[0].iov_len은 25로 설정되어 있습니다.

- 취약점: 입력의 25번째 바이트가 카나리의 **첫 번째 바이트(LSB)**를 덮어씁니다.

- 보통 카나리의 LSB는 \x00입니다. 이를 다른 값으로 덮어쓰면 함수 종료 시(return) __stack_chk_fail이 호출되어 프로그램이 강제 종료됩니다. 이를 통해 공격자는 원하는 시점에 고의로 에러 핸들러를 호출할 수 있습니다.

### B. BSS 영역 임의 쓰기 (The Payload)

변수 v2와 v3가 스택 상에서 iovec 구조체 바로 뒤에 위치하므로, readv(..., 2) 호출 시 이들은 두 번째 iovec 구조체로 해석됩니다.

- Vector 1: _bss_start 주소에 224바이트를 쓸 수 있습니다.

- main 함수에서 _bss_start 주소를 유출해 주었으므로, 공격자는 이 위치가 어디인지 정확히 알고 있습니다.

routine : stdout에 224바이트 (E0)만큼 데이터 쓰기

```python
def swap4(x):
    # Nibble Swap (0xAB -> 0xBA)
    return ((x & 0xF0) >> 4) | ((x & 0x0F) << 4)

# 1. 원본 텍스트 재구성
# 문제의 핵심: 바이너리는 텍스트 파일의 줄바꿈을 \r\n (2바이트)로 인식합니다.
lines = [
    "When I heard the learn’d astronomer,",
    "When the proofs, the figures, were ranged in columns before me,",
    "When I was shown the charts and diagrams, to add, divide, and measure them,",
    "When I sitting heard the astronomer where he lectured with much applause in the lecture-room,",
    "", # 빈 줄
    "How soon unaccountable I became tired and sick,",
    "Till rising and gliding out I wander’d off by myself,",
    "In the mystical moist night-air, and from time to time,",
    "Look’d up in perfect silence at the stars."
]

# join을 이용해 강제로 Windows 스타일 줄바꿈(\r\n)을 적용합니다.
full_text = "\r\n".join(lines)

# 2. UTF-8 바이트 변환 (스마트 따옴표 ’ 가 3바이트로 처리됨)
text_bytes = full_text.encode('utf-8')

# 3. 키(Key) 생성
spaces = []
for idx, byte_val in enumerate(text_bytes):
    if byte_val == 32: # Space
        spaces.append(idx)

g_key = []
for i in range(len(spaces)):
    # Key = (공백 위치 LSB) ^ (7 * i)
    val = (spaces[i] & 0xFF) ^ ((7 * i) & 0xFF)
    g_key.append(val)

# 4. Target Data (IDA 데이터)
target_data = [
    0xD4, 0xC4, 0x74, 0x74, 0xD4, 0xA5, 0x96, 0x44, 0x34, 0x8F, 
    0xF6, 0x84, 0x97, 0x66, 0x05, 0x27, 0x14, 0x66, 0xE6, 0xE6, 0xE6,
    0xF6, 0x77, 0x83, 0x51, 0xF1, 0x31, 0x00, 0xBA, 0xBC, 0x44,
    0x75, 0xC3, 0xC3, 0x63, 0x45, 0xF4, 0x74, 0x55, 0x36, 0x33,
    0x62, 0x93, 0xD2, 0x02, 0x33, 0x4C, 0x1C, 0x32, 0x83, 0x65, 0xA0,
    0x90, 0x83
]

# 5. 복호화
flag = ""
print(f"Text Byte Length: {len(text_bytes)}")
print(f"Key Count: {len(g_key)}")

for i in range(len(target_data)):
    if i >= len(g_key):
        flag += "?"
        continue

    # Decrypt: swap4(encrypted) ^ (i + key[i])
    swapped = swap4(target_data[i])
    xor_mask = (i + g_key[i]) & 0xFF
    dec_char = swapped ^ xor_mask
    
    flag += chr(dec_char)

print("-" * 50)
print(f"FLAG: {flag}")
print("-" * 50)
```

ENTRIES = 256  # table1/table2는 256 엔트리라고 가정 (문제 형태상 거의 확정)

def get_symbol_info(path: str, name: str):
    """Return (st_value, st_size) from .symtab/.dynsym"""
    with open(path, "rb") as f:
        ef = ELFFile(f)
        for secname in (".symtab", ".dynsym"):
            sec = ef.get_section_by_name(secname)
            if not sec:
                continue
            for sym in sec.iter_symbols():
                if sym.name == name:
                    return int(sym["st_value"]), int(sym["st_size"])
    raise KeyError(f"symbol not found: {name}")

def read_vaddr(path: str, vaddr: int, size: int) -> bytes:
    """Read file bytes corresponding to a virtual address (PT_LOAD mapping)."""
    with open(path, "rb") as f:
        ef = ELFFile(f)
        for seg in ef.iter_segments():
            if seg["p_type"] != "PT_LOAD":
                continue
            start = int(seg["p_vaddr"])
            end   = start + int(seg["p_filesz"])
            if start <= vaddr < end and vaddr + size <= end:
                off = int(seg["p_offset"]) + (vaddr - start)
                f.seek(off)
                return f.read(size)
    raise RuntimeError(f"vaddr not mapped in file: {vaddr:#x} (size={size})")

def parse_table(path: str, sym_name: str):
    addr, sz = get_symbol_info(path, sym_name)

    # 테이블 엔트리 수가 256이라는 전제에서 엔트리 크기를 결정
    if sz % ENTRIES != 0:
        raise RuntimeError(f"{sym_name} size not divisible by 256: size={sz}")
    elem_sz = sz // ENTRIES  # 1 또는 4가 정상적

    raw = read_vaddr(path, addr, sz)

    if elem_sz == 1:
        arr = list(raw)  # 0..255
    elif elem_sz == 4:
        arr = list(struct.unpack("<256I", raw))
        arr = [x & 0xFF for x in arr]  # 실제로는 byte 값만 씀
    else:
        raise RuntimeError(f"unexpected elem size for {sym_name}: {elem_sz}")

    return arr

def parse_key(path: str):
    addr, sz = get_symbol_info(path, "key")
    raw = read_vaddr(path, addr, sz)

    # 마지막이 널이면 terminator로 보고 제거 (그 외 널은 그대로 유지해야 함)
    if raw and raw[-1] == 0:
        raw = raw[:-1]
    return raw  # bytes (널 포함 가능)

def solve_one(path: str) -> str:
    table1 = parse_table(path, "table1")
    table2 = parse_table(path, "table2")
    key    = parse_key(path)              # 길이 = 입력 개수 N

    # 안전한 역매핑: out = table1[ table2[x] ] 를 직접 만들어 out->x 후보들을 저장
    inv = {}
    for x in range(ENTRIES):
        outb = table1[ table2[x] ] & 0xFF
        inv.setdefault(outb, []).append(x)

    ans = []
    for b in key:
        if b not in inv:
            raise RuntimeError(f"no preimage for key byte {b:#x}")
        ans.append(inv[b][0])  # 후보 여러 개여도 하나면 통과

    return " ".join(map(str, ans))

def main():
    p = remote(HOST, PORT)

    for r in range(1, 51):
        # 다음 바이너리 수신
        p.recvuntil(b"Binary (base64): ")
        b64_data = p.recvline().strip()

        # 파일 저장
        with open("temp_chall", "wb") as f:
            f.write(base64.b64decode(b64_data))

        payload = solve_one("temp_chall")

        # 정답 전송
        p.sendlineafter(b"integers): ", payload.encode())

        # 오답이면 서버가 바로 Wrong!을 찍고 종료하므로, 한 줄만 체크
        line = p.recvline(timeout=0.5)
        if line:
            if b"Wrong!" in line or b"Challenge failed" in line:
                # 남은 메시지도 뽑아서 원인 확인
                rest = p.recvall(timeout=0.5)
                print((line + rest).decode(errors="ignore"))
                return
            # 라운드 헤더를 미리 읽어버린 경우 대비: 다시 버퍼에 넣기
            if b"Binary (base64):" in line:
                p.unrecv(line)
            # "=== ROUND" 라인이면 그냥 버려도 다음 recvuntil이 정상 동작합니다.

    # 50라운드 끝
    p.interactive()

if __name__ == "__main__":
    main()
